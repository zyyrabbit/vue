export default {
	name: 'DxProgress',
	componentName: 'DxProgress',
	props: {
		percentage: {
			type: Number,
			default: 0,
			required: true,
			validator: val => val >= 0 && val <= 100
		},
		status: {
			type: String
		},
		color: {
			type: String
		},
		strokeWidth: {
			type: Number,
			default: 6
		},
		textInside: {
			type: Boolean,
			default: false
		},
		showText: {
			type: Boolean,
			default: true
		},
		iconClass: {
			type: String,
			default: 'icon iconfont dx-success dx-progress__text--icon'
		}
	},
	computed: {
		barStyle() {
			let style = {}
			style.width = this.percentage + '%'
			style.backgroundColor = this.color
			return style
		},
		progressTextSize() {
			return 12 + this.strokeWidth * 0.4
		}
	}
}
