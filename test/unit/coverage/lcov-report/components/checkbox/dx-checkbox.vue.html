// 用于与组件dx-form-item通信
import Bus from '@/utils/bus.js'
export default {
    name: 'DxCheckbox',
    componentName: 'DxCheckbox',
    props: {
		btnType: Boolean,
		label: {
			type: String,
			required: true
		},
		value: {},
		name: {
			type: String,
			default: 'dx-checkbox'
		},
		disabled: Boolean
    },
    data() {
		return {
			checked: false
		}
    },
    computed: {
		// 找到父元素dx-checkbox-group组件
		parent() {
			let parent = this.$parent
			while (parent && parent.$options.componentName !== 'DxCheckboxGroup') {
				parent = parent.$parent
			}
			return parent
		},
		selectVal() {
			return this.value ? this.value : this.label
		}
    },
    methods: {
		select() {
			if (!this.disabled) {
				this.checked = !this.checked
				// 作为group使用
				if (this.parent) {
					let model = this.parent.model
					if (this.checked) {
						model.push(this.selectVal)
					} else {
						let index = model.indexOf(this.selectVal)
						model.splice(index, 1)
					}
				}
			}
		}
    },
    mounted() {
		Bus.$emit('dx-checkbox-add', this)
    },
    destroyed() {
		Bus.$emit('dx-checkbox-remove', this)
    }
}
