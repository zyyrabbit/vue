<template>
	<div class="dx-notify">
		<transition-group 
			name="msg-list" 
			tag="ul" 
			mode="out-in"
		>
		    <li 
		    	v-for="(msg, index) in msgQueue" 
		    	v-bind:key="msg.key" 
		    	class="dx-notify__msg-list"
		    >
		     	<p class="dx-notify__msg-list--title" >
		     		<span>{{msg.title}}</span>
		     		<span
		     			class="dx-notify__msg-list--close" 
		     			@click="close(msg)" 
		     		> 
		     			x
		     		</span>
		     	</p>
				<p class="dx-notify__msg-list--message" >
					<slot :name="msg.key">
						{{msg.content}}
					</slot>
				</p>
		    </li>
  		</transition-group>
	</div>
</template>
<script>
	export default {
		data() {
			return {
				// 消息队列
				msgQueue: []
			}
		},
		methods: {
			close(message) {
				if (message.timer) {
					clearTimeout(message.timer)
					message.timer = null
				}
				for (let i = message.index + 1; i < this.msgQueue.length; i++) {
					this.msgQueue[i].index--
				}
				this.msgQueue.splice(message.index, 1)
			}
		}
	}
</script>
