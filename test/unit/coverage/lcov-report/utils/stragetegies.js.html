    isNumAndCharAnd_: (rule, value, callback) => {
        if (!/^[\w_]|[\u4E00-\u9FA5]*$/.test(value)) {
            callback && typeof callback === 'function' && callback()
            return false
        }
        return true

    pwdLevelVerification: (rule, value, callback) => {
        let level = 0
        let { min, max } = rule
        // 密码长度
        if ((min && value.length < min) || (max && value.length > max)) {
            return level
        }
        if (/[A-Z]/.test(value)) {
            level++
        }
        if (/[a-z]/.test(value)) {
            level++
        }
        if (/[0-9]/.test(value)) {
            level++
        }
        if (/\W|_/.test(value)) {
            level++
        }
        return level
    }
